            MongoClient.connect('mongodb://아이디:비밀번호@ip:포트번호', (err, database) =>{
                if(err){
                    // 에러 발생했을 때 처리
                }else{
                    // 정상 작동했을 때 처리
                    let db = database.db('데이터베이스 이름');

                }
            });
